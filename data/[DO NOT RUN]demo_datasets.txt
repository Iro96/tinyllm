from datasets import load_dataset, concatenate_datasets, Dataset
import re
import itertools
import random

# ----------------------------
# Text cleaning
# ----------------------------

def clean_text(text: str) -> str:
    text = text.strip()
    text = re.sub(r"\s+", " ", text)
    return text

def length_filter(example, min_chars=50, max_chars=2000):
    return min_chars <= len(example["text"]) <= max_chars


# ----------------------------
# Dataset loaders (SAFE)
# ----------------------------

def load_wikitext():
    ds = load_dataset("wikitext", "wikitext-103-v1", split="train")
    ds = ds.filter(lambda x: x["text"].strip() != "")
    return ds


def load_wikipedia_sample(max_samples=2_000_000):
    ds = load_dataset(
        "wikipedia",
        "20220301.en",
        split="train",
        streaming=True
    )

    ds = (
        ex for ex in ds
        if ex["text"].strip() != ""
    )

    ds = itertools.islice(ds, max_samples)
    return Dataset.from_generator(lambda: ds)


def load_books_small():
    ds = load_dataset("openwebtext", split="train")
    ds = ds.shuffle(seed=42).select(range(1_000_000))
    return ds


def load_code():
    ds = load_dataset("code_search_net", "python", split="train")
    ds = ds.rename_column("code", "text")

    ds = ds.filter(lambda x: 100 <= len(x["text"]) <= 2000)
    return ds


def load_math():
    ds = load_dataset("deepmind/math_dataset", split="train")

    def format_math(ex):
        return {
            "text": f"Question: {ex['question']} Answer: {ex['answer']}"
        }

    ds = ds.map(format_math, remove_columns=ds.column_names)
    ds = ds.shuffle(seed=42).select(range(500_000))
    return ds


# ----------------------------
# Unified loader with weights
# ----------------------------

def load_all_datasets():
    datasets = []

    print("Loading WikiText...")
    datasets.append(load_wikitext())

    print("Loading Wikipedia (sampled)...")
    datasets.append(load_wikipedia_sample())

    print("Loading Books (OpenWebText subset)...")
    datasets.append(load_books_small())

    print("Loading Code...")
    datasets.append(load_code())

    print("Loading Math...")
    datasets.append(load_math())

    # Normalize text
    for i in range(len(datasets)):
        datasets[i] = datasets[i].map(
            lambda x: {"text": clean_text(x["text"])},
            num_proc=4
        )
        datasets[i] = datasets[i].filter(length_filter)

    print("Concatenating datasets...")
    return concatenate_datasets(datasets)


# ----------------------------
# Entry point
# ----------------------------

if __name__ == "__main__":
    dataset = load_all_datasets()
    print(dataset)
    print(dataset[0]["text"][:500])
